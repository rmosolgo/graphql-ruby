def initialize(name, abstract: false, leaf: false, fields: false, wraps: false, input: false, description: nil)
  @name = name
  @abstract = abstract
  @fields = fields
  @wraps = wraps
  @input = input
  @leaf = leaf
  @composite = fields? || abstract?
  @description = description
  freeze
end
