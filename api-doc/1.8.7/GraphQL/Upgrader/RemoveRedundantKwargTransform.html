def apply(input_text)
  if input_text =~ @finder_pattern
    field_name = $~[:name]
    kwarg_value = $~[:kwarg_value]
    if field_name == kwarg_value
      # It's redundant, remove it
      input_text = input_text.sub(/, #{@kwarg}: ['":]#{kwarg_value}['"]?/, "")
    end
  end
  input_text
end
