def call(visit_type, irep_node)
  @memo = @reducer.call(@memo, visit_type, irep_node)
rescue AnalysisError => err
  @errors << err
end

def finalize_reducer
  if @errors.any?
    @errors
  elsif reducer.respond_to?(:final_value)
    reducer.final_value(@memo)
  else
    @memo
  end
end
