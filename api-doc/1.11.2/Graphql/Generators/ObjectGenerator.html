def self.normalize_type_expression(type_expression, mode:, null: true)
  case type_expression
  when "Text"
    ["String", null]
  when "DateTime", "Datetime"
    ["GraphQL::Types::ISO8601DateTime", null]
  when "Date"
    ["GraphQL::Types::ISO8601Date", null]
  else
    super
  end
end
