def self.operation_name
  if (m = Fiber[:__graphql_current_multiplex])
    m.context[:__graphql_current_operation_name] ||= begin
      names = m.queries.map { |q| q.selected_operation_name }
      if names.all?(&:nil?)
        nil
      else
        names.join(",")
      end
    end
  else
    nil
  end
end
