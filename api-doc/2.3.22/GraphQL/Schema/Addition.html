def initialize(schema:, own_types:, new_types:)

  @own_types = own_types
  @directives = Set.new
  @possible_types = {}
  @types = {}
  @union_memberships = {}
  @references = Hash.new { |h, k| h[k] = Set.new }
  @arguments_with_default_values = []
  add_type_and_traverse(new_types)
end
