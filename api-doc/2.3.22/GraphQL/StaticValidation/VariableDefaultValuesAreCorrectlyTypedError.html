def initialize(message, path: nil, nodes: [], name:, type: nil, error_type:)
  super(message, path: path, nodes: nodes)
  @variable_name = name
  @type_name = type
  raise("Unexpected error type: #{error_type}") if !VIOLATIONS.values.include?(error_type)
  @violation = error_type
end
