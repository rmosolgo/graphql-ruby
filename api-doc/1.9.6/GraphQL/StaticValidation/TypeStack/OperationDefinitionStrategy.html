def push(stack, node)
  # eg, QueryType, MutationType
  object_type = stack.schema.root_type_for_operation(node.operation_type)
  stack.object_types.push(object_type)
  stack.path.push("#{node.operation_type}#{node.name ? " #{node.name}" : ""}")
end
