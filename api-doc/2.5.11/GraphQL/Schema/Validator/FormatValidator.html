def validate(_object, _context, value)
  if permitted_empty_value?(value)
    # Do nothing
  elsif value.nil? ||
      (@with_pattern && !value.match?(@with_pattern)) ||
      (@without_pattern && value.match?(@without_pattern))
    @message
  end
end
