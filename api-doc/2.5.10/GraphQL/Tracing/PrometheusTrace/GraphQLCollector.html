def collect(object)
  default_labels = { key: object['key'], platform_key: object['platform_key'] }
  custom = object['custom_labels']
  labels = custom.nil? ? default_labels : default_labels.merge(custom)

  @graphql_gauge.observe object['duration'], labels
end
