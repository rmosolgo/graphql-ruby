def get_references_to(type_name)
  own_refs = @own_references_to[type_name]
  inherited_refs = superclass.references_to(type_name)
  if inherited_refs&.any?
    if own_refs&.any?
      own_refs + inherited_refs
    else
      inherited_refs
    end
  else
    own_refs
  end
end
