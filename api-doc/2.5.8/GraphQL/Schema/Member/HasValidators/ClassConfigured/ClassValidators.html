def validators
  inherited_validators = superclass.validators
  if !inherited_validators.empty?
    if @own_validators.nil?
      inherited_validators
    else
      inherited_validators + @own_validators
    end
  elsif @own_validators.nil?
    EMPTY_ARRAY
  else
    @own_validators
  end
end
