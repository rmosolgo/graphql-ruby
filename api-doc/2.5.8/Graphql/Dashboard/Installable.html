def check_installed
  if !feature_installed?
    @component_header_html = self.class::INSTALLABLE_COMPONENT_HEADER_HTML
    @component_message_html = self.class::INSTALLABLE_COMPONENT_MESSAGE_HTML
    render "graphql/dashboard/not_installed"
  end
end
