def add_error(error)
  if !error.is_a?(ExecutionError)
    raise TypeError, "expected error to be a ExecutionError, but was #{error.class}"
  end
  errors << error
  nil
end

def spawn_child(key:, irep_node:, object:)
  FieldResolutionContext.new(
    context: @context,
    parent: self,
    object: object,
    key: key,
    irep_node: irep_node,
  )
end
