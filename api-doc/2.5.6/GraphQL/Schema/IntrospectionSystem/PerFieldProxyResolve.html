def call(obj, args, ctx)
  query_ctx = ctx.query.context
  # Remove the QueryType wrapper
  if obj.is_a?(GraphQL::Schema::Object)
    obj = obj.object
  end
  wrapped_object = @object_class.wrap(obj, query_ctx)
  @inner_resolve.call(wrapped_object, args, ctx)
end
