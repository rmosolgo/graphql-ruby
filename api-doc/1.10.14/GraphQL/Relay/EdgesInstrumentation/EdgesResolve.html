def call(obj, args, ctx)
  parent = ctx.object
  nodes = @resolve_proc.call(obj, args, ctx)
  if ctx.schema.lazy?(nodes)
    nodes
  else
    nodes.map { |item| item.is_a?(GraphQL::Pagination::Connection::Edge) ? item : @edge_class.new(item, parent) }
  end
end
