def visible?(context)
  if dir = self.directives.find { |d| d.is_a?(Flagged) }
    relevant_flags = (f = context[:flags]) && dir.arguments[:by] & f
    relevant_flags && relevant_flags.any? && super
  else
    super
  end
end
