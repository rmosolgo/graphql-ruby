def relation_count(relation)
  int_or_hash = if relation.respond_to?(:unscope)
    relation.unscope(:order).count(:all)
  else
    # Rails 3
    relation.count
  end
  if int_or_hash.is_a?(Integer)
    int_or_hash
  else
    # Grouped relations return count-by-group hashes
    int_or_hash.length
  end
end
